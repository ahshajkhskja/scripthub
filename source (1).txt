local function createFrame(size, pos)
    local f = Instance.new("Frame")
    f.Size = size
    f.Position = pos
    f.BackgroundColor3 = Color3.fromRGB(15, 15, 15)
    f.BackgroundTransparency = 0.25
    f.ClipsDescendants = true
    Instance.new("UICorner", f).CornerRadius = UDim.new(0, 8)
    local s = Instance.new("UIStroke", f)
    s.Color = Color3.fromRGB(120, 120, 120)
    s.Thickness = 1.2
    f.Parent = sg
    return f
end

local dragging, dragInput, dragStart, startPos
local function updateInput(input)
    local delta = input.Position - dragStart
    main.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
end
main.InputBegan:Connect(function(inp)
    if inp.UserInputType == Enum.UserInputType.MouseButton1 or inp.UserInputType == Enum.UserInputType.Touch then
        dragging = true
        dragStart = inp.Position
        startPos = main.Position
        inp.Changed:Connect(function()
            if inp.UserInputState == Enum.UserInputState.End then dragging = false end
        end)
    end
end)
main.InputChanged:Connect(function(inp)
    if inp.UserInputType == Enum.UserInputType.MouseMovement or inp.UserInputType == Enum.UserInputType.Touch then
        dragInput = inp
    end
end)
UserInputService.InputChanged:Connect(function(inp)
    if dragging and inp == dragInput then updateInput(inp) end
end)
local function slide(obj, prop)
    TweenService:Create(obj, TweenInfo.new(0.4, Enum.EasingStyle.Quart, Enum.EasingDirection.Out), prop):Play()
end
toggle.MouseButton1Click:Connect(function()
    if main.Position.Y.Scale > 1 then
        slide(main, {Position = UDim2.new(0.5, -105, 0.93, -65)})
    else
        slide(main, {Position = UDim2.new(0.5, -105, 1.5, 0)})
        hackerPanel.Visible = false
    end
end)
local function updateSearch(query)
    for _, v in pairs(listScroll:GetChildren()) do if v:IsA("TextButton") then v:Destroy() end end
    if query == "" then
        slide(hackerPanel, {Size = UDim2.new(0, 0, 0, 200)})
        task.delay(0.3, function() hackerPanel.Visible = false end)
        return
    end
    hackerPanel.Visible = true
    local maxWidth = 240
    local matches = {}
    for _, s in pairs(scripts) do
        if s[1]:lower():find(query:lower()) then
            table.insert(matches, s)
        end
    end
    for _, s in pairs(matches) do
        local btn = Instance.new("TextButton", listScroll)
        btn.Size = UDim2.new(1, 0, 0, 24)
        btn.BackgroundTransparency = 0.8
        btn.BackgroundColor3 = Color3.new(0.2, 0.2, 0.2)
        btn.Font = Enum.Font.Code
        btn.TextColor3 = Color3.new(1, 1, 1)
        btn.TextSize = 12
        btn.Text = " " .. s[1]
        btn.TextXAlignment = Enum.TextXAlignment.Left
        Instance.new("UICorner", btn)
        btn.MouseButton1Click:Connect(function()
            task.spawn(function() loadstring(game:HttpGet(s[2]))() end)
        end)
    end
    slide(hackerPanel, {Size = UDim2.new(0, maxWidth, 0, 200)})
end
input:GetPropertyChangedSignal("Text"):Connect(function()
    updateSearch(input.Text)
end)
input.FocusLost:Connect(function(enter)
    if not enter then return end
    local val = input.Text:lower()
    for _, s in pairs(scripts) do
        if s[1]:lower():find(val) then
            task.spawn(function() loadstring(game:HttpGet(s[2]))() end)
            break
        end
    end
    input.Text = ""
    updateSearch("")
end)
